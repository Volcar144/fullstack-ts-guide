# TypeScript in React

## Typing props

```tsx
// Interface approach (recommended for components)
interface CardProps {
  title: string;
  description: string;
  tags?: string[];
  onSelect: (title: string) => void;
}

function Card({ title, description, tags = [], onSelect }: CardProps) {
  return (
    <div onClick={() => onSelect(title)}>
      <h3>{title}</h3>
      <p>{description}</p>
    </div>
  );
}
```

## Typing useState

```tsx
const [name, setName] = useState<string>(''); // explicit
const [name, setName] = useState('');          // inferred — same thing

// When the type isn't obvious from initial value
const [user, setUser] = useState<User | null>(null);
const [items, setItems] = useState<string[]>([]);
```

## Typing event handlers

```tsx
function Form() {
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    console.log(e.target.value);
  };

  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    // ...
  };

  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    // ...
  };

  return (
    <form onSubmit={handleSubmit}>
      <input onChange={handleChange} />
      <button onClick={handleClick}>Submit</button>
    </form>
  );
}
```

## Typing useRef

```tsx
import { useRef } from 'react';

// DOM reference
const inputRef = useRef<HTMLInputElement>(null);

// Focus the input
const focusInput = () => {
  inputRef.current?.focus(); // ?. because it might be null
};

// Mutable value reference (no re-render)
const timerRef = useRef<ReturnType<typeof setInterval>>(null);
```

## Common React types

```tsx
// JSX you can render
React.ReactNode      // string, number, JSX, arrays, null
React.ReactElement   // JSX only

// Component type (function component)
React.FC<Props>      // avoid — causes issues with children
(props: Props) => JSX.Element  // prefer this

// Styles
React.CSSProperties  // for inline style prop

// Refs
React.RefObject<HTMLDivElement>
React.MutableRefObject<number>
```

## A fully-typed component example

```tsx
interface SearchInputProps {
  value: string;
  onChange: (value: string) => void;
  placeholder?: string;
  onClear?: () => void;
}

export function SearchInput({
  value,
  onChange,
  placeholder = 'Search...',
  onClear,
}: SearchInputProps) {
  return (
    <div className="relative">
      <input
        type="search"
        value={value}
        onChange={(e: React.ChangeEvent<HTMLInputElement>) => onChange(e.target.value)}
        placeholder={placeholder}
        className="w-full rounded-lg border px-4 py-2 pr-10"
      />
      {value && onClear && (
        <button
          onClick={onClear}
          className="absolute right-3 top-1/2 -translate-y-1/2 text-gray-400"
          aria-label="Clear search"
        >
          ×
        </button>
      )}
    </div>
  );
}
```
